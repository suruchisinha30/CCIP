#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    
    int arr[20],i;
    int n;
    cin>>n;
    int flag=0;
    for(int i=0 ; i<n ; i++){
        cin>>arr[i];
        
    }
    for(int i=0 ; i<=n/2 ; i++){
        if(arr[i]!=arr[n-i-1]){
            flag=1;
            break;
        }
    }
    if(flag==1){
        cout<<"Not Verified";
    }
    else{
        cout<<"Verified";
    }
     return 0;
}
