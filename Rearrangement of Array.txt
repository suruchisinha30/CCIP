#include <iostream>
using namespace std;
int main()
{
    int Arr[20],n,i;
    
    cin>>n;

    for(i=0;i<n;i++)
    {
        cin>>Arr[i];
    }
    
    int temp = Arr[0];
    for(i=0;i<n-1;i++)
    {
      Arr[i] = Arr[i+1];
    }
    Arr[n-1] = temp; 
    for(i=0;i<n;i++)
        cout<<Arr[i]<< " ";
    return 0;
}
