#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;


int main() {
    /* Enter your code here. Read input from STDIN. Print output to STDOUT */  
    int arr[6],i, mul=0,sum=0;
    for(i=1; i<=6; i++)
    {
        cin>>arr[i];
    }
    for(i=1; i<=6; i++)
    {
        mul= arr[i] * arr[i+1];
        sum= sum + mul;
        i++;
    }
    
    cout<<sum;
    
    return 0;
}
