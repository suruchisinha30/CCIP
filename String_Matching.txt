#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;


int main() {
    /* Enter your code here. Read input from STDIN. Print output to STDOUT */   
    
    int flag=1;
    string str1;
    string str2;
    cin>>str1;
    cin>>str2;
    if(str1.length()==str2.length()){
        for (int i = 0; i < 3; i++){
        if (str1[i] == str2[i])
           flag=1;
        }
        
        if(flag==1)
           cout<< "Yes"; 
        else
            cout<<"No";
    }
    else
        cout<<"No";
    return 0;
}
